      body = `<div class="cands">` + tk.cands.map((c,ci)=>`

        <div class="cand ${c.pick?'pick':''}" data-track="${tk.i}" data-ci="${ci}">

          <span class="radio"></span>

          <div class="cinfo"><div class="c1">${c.c1}</div><div class="c2">${c.c2}</div></div>

          <span class="cdur">${fmt(c.dur)}</span>

        </div>`).join('') + `</div>

        <div class="reason"><span class="qi">&#9672;</span><b>${tk.reason}</b></div>`;

        No extended version exists &mdash; the radio edit is the only upload found. Skipped from the playlist.

      </div>`;

    }

    <div class="track ${tk.state}" data-i="${tk.i}">

  const reduce = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

  const qfill = document.getElementById('qfill');

  const qnum = document.getElementById('qnum');

  const worker = document.getElementById('worker');

  const wnow = document.getElementById('wnow');

  const wcount = document.getElementById('wcount');


  function fillBars(el){

    el.querySelectorAll('.bar').forEach(b=>{

      const w = b.getAttribute('data-w');

      if(w!==null) requestAnimationFrame(()=> b.style.width = w + '%');

    });

    const dt = el.querySelector('.delta-tag');

    if(dt) setTimeout(()=> dt.classList.add('on'), 300);

  }

  function setQuota(units){

    qnum.textContent = units.toLocaleString();

    qfill.style.width = Math.min(100,(units/10000)*100) + '%';

  }

  function run(){

    const els = [...document.querySelectorAll('.track')];

    els.forEach(e=>{ e.classList.remove('shown'); e.querySelectorAll('.bar').forEach(b=>b.style.width='0%'); e.querySelectorAll('.delta-tag').forEach(d=>d.classList.remove('on')); });

    setQuota(0);

    worker.classList.add('live');


    if(reduce){

      els.forEach(e=>{ e.classList.add('shown'); fillBars(e); });

      setQuota(840); wnow.textContent='6 of 6 done'; wcount.textContent='5 / 6 matched';

      worker.classList.remove('live'); return;

    }


    let units = 0, done = 0, matched = 0;

    els.forEach((e,idx)=>{

      setTimeout(()=>{

        e.classList.add('shown'); fillBars(e); done++;

        units += 100 + (tracks[idx].state==='review'?100:0);

        setQuota(units);

        if(tracks[idx].state==='strong'||tracks[idx].state==='review') matched++;

        wnow.textContent = done<els.length ? `track ${done+1} of ${els.length}` : `${els.length} of ${els.length} done`;

        wcount.textContent = `${matched} / ${els.length} matched`;

        if(done===els.length){ setTimeout(()=> worker.classList.remove('live'), 400); }

      }, 350 + idx*520);

    });

  }

    const cand = e.target.closest('.cand');

    if(!cand) return;

    cand.closest('.cands').querySelectorAll('.cand').forEach(c=>c.classList.remove('pick'));

    cand.classList.add('pick');

  });

  }

  list.addEventListener('change', e=>{ if(e.target.classList.contains('incbox')) updateSel(); });

    setTimeout(()=> toast.classList.remove('on'), 3800);

  });

  document.getElementById('find').addEventListener('click', run);
